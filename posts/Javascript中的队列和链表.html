队列,先进先出.排过队吗?按顺序添加和处理的任务,都可以用队列的结构进行存储和消费.

1class Queue {
2  constructor () {
3    this._items = {}
4    this._count = 0
5    this._lowestCount = 0
6  }
7 
8  enqueue(e) {
9    this._items[this._count] = e
10    this._count++
11  }
12  dequeue() {
13    if(this.isEmpty()) return undefined
14    const r = this._items[this._lowestCount]
15    delete this._items[this._lowestCount]
16    this._lowestCount += 1
17    return r
18  }
19 
20  isEmpty() {
21    return this._lowestCount === this._count
22  }
23  peek() {
24    return this.isEmpty()  ? undefined : this._items[this._lowestCount]
25  }
26 
27  size() {
28    return this._count - this._lowestCount
29  }
30 
31  clear() {
32    this._items = {}
33    this._count = 0
34    this._lowestCount = 0
35  }
36  toString() {
37    if(this.isEmpty()) return ''
38    let r = ''
39    for (const iterator of Object.values(this._items)) {
40      r += r === '' ? `${iterator}` : `, ${iterator}`
41    }
42    return r
43  }
44}
45
