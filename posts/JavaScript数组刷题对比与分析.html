      /**
 * @param {number[]} A
 * @param {number} K
 * @return {number[]}
 */
var addToArrayForm = function (A, K) {
  let kArr = [...K.toString()].map(i => ~~i)
  let {len, maxArr} = kArr.length > A.length ? {len: kArr.length, maxArr: kArr} : {len: A.length, maxArr: A}
  let carryNum = 0
  for (let index = 0; index < len; index++) {
    let lastA = ~~A.pop()
    let lastK = ~~kArr.pop()
    let sum = lastA + lastK + carryNum
    if(sum > 9) {
      maxArr.unshift(sum - 10)
      carryNum = 1
    } else {
      maxArr.unshift(sum)
      carryNum = 0
    }
  }
  if(carryNum === 1) maxArr.unshift(1)
  return maxArr
}

var addToArrayForm = function (A, K) {
    count = 0;
    len = K.toString().length;
    for (let i = 0; i < len; i++) {
        if (A.length < len) {
            A.unshift(0);
        }
        count = K % 10;
        K = parseInt(K / 10);
        A[A.length - 1 - i] += count;
    }
    for (let i = 0; i < A.length; i++) {
        if (A[0] > 9) {
            A.unshift(0);
        }
        if (A[A.length - 1 - i] > 9) {
            A[A.length - 2 - i]++;
            A[A.length - 1 - i] = A[A.length - 1 - i] % 10;
        }
    }
    return A;
};
